RTE Act Information

Under the Right to Education Act, children aged 6-14 are entitled to free and compulsory education in government schools. Private schools must also reserve 25% seats for economically weaker sections.

const schoolType = document.getElementById('schoolType').value; const gradeLevel = document.getElementById('gradeLevel').value; const services = Array.from(document.querySelectorAll('input[name="services"]:checked')).map(cb => cb.value); // Calculate base tuition fee let tuitionFee = 0; switch(schoolType) { case 'government': tuitionFee = 1000; break; case 'private': tuitionFee = 5000; break; case 'international': tuitionFee = 10000; break; } // Apply grade level multiplier const gradeMultiplier = { 'primary': 1, 'middle': 1.2, 'secondary': 1.5, 'senior': 1.8 }; tuitionFee *= gradeMultiplier[gradeLevel]; // Calculate additional services fee let servicesFee = 0; if(services.includes('transport')) servicesFee += 2000; if(services.includes('meals')) servicesFee += 1500; if(services.includes('activities')) servicesFee += 1000; // Update the result section document.getElementById('tuitionFee').textContent = `₹${Math.round(tuitionFee)}`; document.getElementById('servicesFee').textContent = `₹${servicesFee}`; document.getElementById('totalFee').textContent = `₹${Math.round(tuitionFee + servicesFee)}`; document.getElementById('result').classList.remove('hidden'); });
